```
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include<stdlib.h>

int main(void)
{
	int x;
	char grade;
	if (scanf("%d", &x) != 1) {
		printf("输入无效,请输入一个整数\n");
		return 1;
	}
	

	//退出程序并返回错误状态
	if (x < 0 || x>100) {
		exit(1);
	}

	if (x >= 90){
		grade = 'A';
	}
	else if (x >= 80 && x <= 89) {
		grade = 'B';
	}
	else if (x >= 70 && x <= 79) {
		grade = 'C';
	}
	else if (x >= 60 && x <= 69) {
		grade = 'D';
	}
	else {
		grade = 'E';
	}
	
	printf("%c", grade);

	return 0;
}
```
