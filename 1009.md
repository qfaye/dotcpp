```
#include<stdio.h>
#include<string.h>
int main(void) {
    char str[10];
    scanf("%s", &str);
    int len = strlen(str);

    printf("%d\n", len);
    for (int i = 0; i < len; i++) {
        printf("%c", str[i]);
        if (i != len - 1)    printf(" ");    // i == len - 1 时即为最后一个数字，不输出空格
    }
    printf("\n");
    for (int i = len - 1; i >= 0; i--) {
        printf("%c", str[i]);
    }
    return 0;
}
```
