#include<stdio.h>

int max_Divisor(int a,int b){

  int c=a%b;
  if(c==0){
    return b;
  }else{
    return max_Divisor(b,c);
  }
}

int min_multiple(int a, int b){
  int max=max_Divisor(a,b);
  return (a*b)/max;
}

int main(void){
  int m,n;
  scanf("%d %d",&m,&n);
  int maxyue=max_Divisor(m,n);

  int minbei=min_multiple(m,n);
  
  printf("%d %d",maxyue,minbei);
  return 0;
}
