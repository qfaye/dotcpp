```
#include<stdio.h>
#include<string.h>

int main(void)
{
    char str[200];
    int letter=0, num=0, space=0, other = 0;;
    //gets(str);//字符串输入
    fgets(str, sizeof(str), stdin);//更安全,防止溢出
    for (int i = 0; i < strlen(str); i++) {
        if (str[i] >= 'a' && str[i] <= 'z' || str[i] >= 'A' && str[i] <= 'Z') {
            letter++;
        }
        else if (str[i] >= '0' && str[i] <= '9') {
            num++;
        }
        else if (str[i] == ' ') {
            space++;
        }
        else {
            other++;
        }
    }
        printf("%d %d %d %d", letter, num, space, other);
        return 0;
    }
```
