```
#include<stdio.h>
#include<math.h>
int main(void) {
    int n, result=0;
    scanf("%d", &n);

    for (int i = n; i > 0; i--) {
        result = result + (pow(10, i) - 1) * 2 / 9;
    }
    printf("%d", result);
    return 0;
}
```
